285impl SizeDistribution {
286    fn validate(&self, kpc: &KappaConfig) -> Result<(), KappaError> {
287        match self {
288            SizeDistribution::Apow if kpc.apow < 0.0 => Err(anyhow!("UnexpectedApow").into()),
289            SizeDistribution::Normal => {
290                if kpc.amean <= 0.0 {
291                    return Err(
292                        anyhow!("`amean` must be positive for (log-)normal distribution").into(),
293                    );
294                }
295                if kpc.asigma == 0.0 {
296                    return Err(
297                        anyhow!("`asigma` cannot be zero for (log-)normal distribution").into(),
298                    );
299                }
300                if kpc.asigma > 0.0 {
301                    return Err(KappaError::ForceLogNormal);
302                }
303                Ok(())
304            }
305            _ => Ok(()),
306        }
307    }
308}

429/// Run the simulation.
430pub fn run(kpc: &mut KappaConfig) -> Result<(), KappaError> {
431    prepare_inputs(kpc)?;
432
433    for material in &mut kpc.materials {
434        match material.cmd {
435            RefractiveIndexKind::CmdLine => {}
436            RefractiveIndexKind::File | RefractiveIndexKind::Other => {
437                let component = get_lnk_data(&material.key);
438                (material.re, material.im) =
439                    regrid_lnk_data(component.l0, component.n0, component.k0, &kpc.lam, true);
440                material.rho = component.rho;
441            }
442        }
443    }
444
445    // Normalize the mass fractions
446    (kpc.tot_mfrac_core, kpc.tot_mfrac_mantle) =
447        kpc.materials
448            .iter()
449            .fold((0.0, 0.0), |(tot_core, tot_mantle), m| match m.kind {
450                MaterialKind::Core => (tot_core + m.mfrac, tot_mantle),
451                MaterialKind::Mantle => (tot_core, tot_mantle + m.mfrac),
452            });
453    kpc.materials.iter_mut().for_each(|m| match m.kind {
454        MaterialKind::Core => {
455            m.mfrac /= kpc.tot_mfrac_core;
456        }
457        MaterialKind::Mantle => {
458            m.mfrac /= kpc.tot_mfrac_mantle;
459        }
460    });
461    kpc.materials.iter_mut().for_each(|m| {
462        m.vfrac = m.mfrac / m.rho;
463    });
464    (kpc.tot_vfrac_core, kpc.tot_vfrac_mantle) =
465        kpc.materials
466            .iter()
467            .fold((0.0, 0.0), |(tot_core, tot_mantle), m| match m.kind {
468                MaterialKind::Core => (tot_core + m.vfrac, tot_mantle),
469                MaterialKind::Mantle => (tot_core, tot_mantle + m.vfrac),
470            });
471
472    // Turn mass fractions into volume fractions
473    kpc.rho_core = kpc.tot_mfrac_core / kpc.tot_vfrac_core;
474    kpc.rho_mantle = kpc.tot_mfrac_mantle / kpc.tot_vfrac_mantle;
475
476    // Normalize volume fractions
477    kpc.materials.iter_mut().for_each(|m| match m.kind {
478        MaterialKind::Core => m.vfrac /= kpc.tot_vfrac_core,
479        MaterialKind::Mantle => m.vfrac /= kpc.tot_vfrac_mantle,
480    });
481
482    if kpc.pcore > 0.0 {
483        kpc.materials.iter_mut().for_each(|m| {
484            if m.kind == MaterialKind::Core {
485                m.vfrac *= 1.0 - kpc.pcore;
486            }
487        });
488        kpc.rho_core *= 1.0 - kpc.pcore;
489    }
490
491    if kpc.pmantle > 0.0 {
492        kpc.materials.iter_mut().for_each(|m| {
493            if m.kind == MaterialKind::Mantle {
494                m.vfrac *= 1.0 - kpc.pmantle;
495            }
496        });
497        kpc.rho_mantle *= 1.0 - kpc.pmantle;
498    }
499
500    // Calculate average density of the whole grain
501    if kpc.nmant == 0 {
502        kpc.rho_av = kpc.rho_core;
503    } else {
504        kpc.rho_av =
505            kpc.rho_core / (1.0 + kpc.tot_mfrac_mantle * (kpc.rho_core / kpc.rho_mantle - 1.0));
506        kpc.tot_vfrac_mantle = kpc.tot_mfrac_mantle * kpc.rho_av / kpc.rho_mantle;
507    }
508
509    // Loop for splitting the output into files by grain size
510    if kpc.split {
511        let mut nsub = kpc.nsubgrains;
512        if nsub % 2 == 0 {
513            nsub += 1;
514        }
515        let afact = (kpc.amax / kpc.amin).powf(1.0 / kpc.na as f64);
516        let afsub = afact.powf(1.0 / (nsub - 1) as f64);
517
518        let bar = Arc::new(ProgressBar::new(kpc.nlam as u64));
519
520        bar.set_style(
521            ProgressStyle::default_bar()
522                .template("{spinner} [{elapsed_precise}] [{bar:40.cyan/blue}] {pos}/{len} ({eta})")
523                .unwrap()
524                .progress_chars("█▓▒░ "),
525        );
526
527        (0..kpc.na).into_par_iter().try_for_each(|ia| {
528            let iaf = ia as f64;
529            let asplit = kpc.amin * afact.powf(iaf + 0.5);
530            let nsubf = nsub as f64;
531            let aminsplit = asplit * afsub.powf(-nsubf / 2.0);
532            let amaxsplit = asplit * afsub.powf(nsubf / 2.0);
533            let _ = compute_kappa(ia, aminsplit, amaxsplit, kpc)?;
534            bar.inc(1); // Increment the progress bar inside the parallel loop
535            Ok::<(), KappaError>(())
536        })?;
537        bar.finish_with_message("Processing complete!");
538        exit(0);
539    } else {
540        let p = compute_kappa(0, kpc.amin, kpc.amax, kpc)?;
541        if !p.is_ok {
542            if kpc.mmf_ss {
543                eprintln!("WARNING: opacities OK, but some F_nn,g_asym may not be accurate");
544            } else {
545                eprintln!("WARNING: opacities OK, but some F_nn,g_asym are set to zero");
546            }
547        }
548
549        // TODO: FITS writer will be implemented later
550
551        write_opacities(kpc, &p)?;
552    }
553
554    Ok(())
555}

557/// Pre-process the inputs collected from the command-line
558fn prepare_inputs(kpc: &mut KappaConfig) -> Result<()> {
559    // Materials
560    if kpc.nmat >= 20 {
561        return Err(anyhow!("TooManyMaterials"));
562    }
563    if kpc.nmat == kpc.nmant && kpc.nmant > 0 {
564        return Err(anyhow!("NoCoreMaterial"));
565    }
566
567    // Porosity
568    if kpc.pcore < 0.0 || kpc.pcore >= 1.0 || kpc.pmantle < 0.0 || kpc.pmantle >= 1.0 {
569        return Err(anyhow!("InvalidPorosity"));
570    }
571
572    // Grain size distribution
573    if kpc.amin <= 0.0 || kpc.amax <= 0.0 {
574        return Err(anyhow!("InvalidSizeInput"));
575    }
576    if kpc.amin >= kpc.amax {
577        swap(&mut kpc.amin, &mut kpc.amax);
578    }
579    if kpc.na == 0 {
580        kpc.na = (((kpc.amax.log10() - kpc.amin.log10()) * 15.0 + 1.0) as usize).max(5);
581    }
582    if kpc.amin == kpc.amax && kpc.na != 1 {
583        kpc.na = 1;
584    }
585    match kpc.sizedis.validate(kpc) {
586        Ok(()) => {}
587        Err(KappaError::ForceLogNormal) => {
588            kpc.sizedis = SizeDistribution::LogNormal;
589        }
590        Err(_) => return Err(anyhow!("InvalidSizeParam")),
591    }
592
593    // Wavelength grid
594    if kpc.lmin <= 0.0 || kpc.lmax <= 0.0 {
595        return Err(anyhow!("InvalidWavelengthInput"));
596    }
597    if kpc.lmin > kpc.lmax {
598        swap(&mut kpc.lmin, &mut kpc.lmax);
599    }
600    if kpc.nlam <= 1 && kpc.lmin != kpc.lmax {
601        return Err(anyhow!("SamplingRequired"));
602    }
603    if kpc.lmin == kpc.lmax && kpc.nlam != 1 {
604        kpc.nlam = 1
605    }
606    if kpc.nsparse > 0 {
607        println!("Creating sparse scattering matrix file");
608    }
609
610    // DHS
611    match kpc.method {
612        KappaMethod::DHS => {
613            if kpc.fmax < 0.0 || kpc.fmax >= 1.0 {
614                return Err(anyhow!("InvalidArgument: fmax"));
615            }
616        }
617        KappaMethod::MMF => {
618            if kpc.mmf_struct > 3.0 {
619                return Err(anyhow!("`mmf_struct` must be between 1 and 3"));
620            }
621            if kpc.mmf_struct <= 0.0 {
622                return Err(anyhow!("`mmf_struct` must be positive"));
623            }
624            if kpc.mmf_a0 >= kpc.amin {
625                return Err(anyhow!("`mmf_a0` must be smaller than `amin`"));
626            }
627        }
628        KappaMethod::CDE => {
629            if kpc.lmin <= 2.0 * PI * kpc.amax {
630                eprintln!("WARNING: CDE requires Rayleigh limit!");
631            }
632        }
633    }
634
635    // Angular grid
636    if kpc.nang % 2 == 1 {
637        return Err(anyhow!("`nang` must be even"));
638    }
639
640    // Other
641    if kpc.split && kpc.blend_only {
642        eprintln!("WARNING: Turning off `-s` for `-blend_only`");
643        kpc.split = false;
644    }
645    if kpc.split && kpc.sizedis != SizeDistribution::Apow {
646        return Err(anyhow!(
647            "`split` is only supported for powerlaw size distribution"
648        ));
649    }
650
651    // Sort by `kind = MaterialKind::Core`
652    kpc.materials.sort_by(|a, b| match (&a.kind, &b.kind) {
653        (MaterialKind::Core, MaterialKind::Core) => Ordering::Equal,
654        (MaterialKind::Core, _) => Ordering::Less,
655        (_, MaterialKind::Core) => Ordering::Greater,
656        _ => Ordering::Equal,
657    });
658
659    kpc.nmat = kpc.ncore + kpc.nmant;
660
661    // Make logarithmic wavelength grid
662    match kpc.wavelength_kind {
663        WavelengthKind::Other => {
664            kpc.lam = RVector::logspace(10.0, kpc.lmin.log10(), kpc.lmax.log10(), kpc.nlam);
665        }
666        WavelengthKind::File => {
667            // already read from file
668        }
669    }
670
671    // Prepare sparse scattering file
672    if kpc.nsparse > 0 {
673        prepare_sparse(kpc);
674    }
675    // Writing wavelength grid file
676    if kpc.write_grid {
677        write_wavelength_grid(kpc)?;
678    }
679
680    Ok(())
681}

683/// Calculate the opacities for a grain-size value.
684fn compute_kappa(
685    ia: usize,
686    amin: f64,
687    amax: f64,
688    kpc: &KappaConfig,
689) -> Result<Particle, KappaError> {
690    let ns = kpc.na;
691    let (nf, ifmn) = if kpc.fmax == 0.0 { (1, 1) } else { (20, 12) };
692
693    let mut r = RVector::zeros(ns);
694    let mut nr = RVector::zeros(ns);
695
696    let mut e1mantle = RVector::zeros(kpc.nlam);
697    let e2mantle = RVector::zeros(kpc.nlam);
698
699    let aminlog = amin.log10();
700    let amaxlog = amax.log10();
701    let pow = -kpc.apow;
702
703    let mut tot = 0.0;
704
705    if ns == 1 {
706        // Just one size
707        r[0] = 10.0_f64.powf((aminlog + amaxlog) / 2.0);
708        nr[0] = r[0].powf(pow + 1.0); // should be 1/r[0]^3 ???  Not important.
709    } else {
710        // Size distribution
711        let nsf = (ns - 1) as f64;
712        for (is, (r_val, nr_val)) in r.iter_mut().zip(nr.iter_mut()).enumerate() {
713            let isf = is as f64;
714            *r_val = 10.0f64.powf(aminlog + (amaxlog - aminlog) * isf / nsf);
715            // Apply size distribution and exponential calculation
716            *nr_val = match kpc.sizedis {
717                // Power law size distribution
718                SizeDistribution::Normal | SizeDistribution::LogNormal => {
719                    if kpc.asigma < 0.0 {
720                        let expo = 0.5 * ((*r_val - kpc.amean) / kpc.asigma).powf(2.0);
721                        if expo > 99.0 {
722                            0.0
723                        } else {
724                            (-expo).exp()
725                        }
726                    } else {
727                        let expo = 0.5 * ((*r_val / kpc.amean).ln() / kpc.asigma).powf(2.0);
728                        if expo > 99.0 {
729                            0.0
730                        } else {
731                            (-expo).exp()
732                        }
733                    }
734                }
735                SizeDistribution::File => {
736                    todo!() // TODO: Move this out of else block
737                }
738                _ => r_val.powf(pow + 1.0),
739            };
740            // With the option `-d`m each computation is only a piece of the size grid
741            if *r_val < amin || *r_val > amax {
742                *nr_val = 0.0;
743            }
744            // Volume normalization
745            tot += *nr_val * r_val.powi(3);
746        }
747        // Normalize nr
748        nr = 1.0 * nr / tot;
749    }
750
751    if kpc.write_grid && ia == 0 {
752        write_sizedis_file(kpc, ns, &r, &mut nr, tot)?;
753    }
754
755    // Copy the refractory index data for all materials into local arrays
756    let mut e1 = RMatrix::zeros((kpc.nlam, kpc.nmat));
757    let mut e2 = RMatrix::zeros((kpc.nlam, kpc.nmat));
758    for im in 0..kpc.nmat {
759        e1.slice_mut(s![.., im]).assign(&kpc.materials[im].re);
760        e2.slice_mut(s![.., im]).assign(&kpc.materials[im].im);
761    }
762
763    let mut e1_blend = RVector::zeros(kpc.nlam);
764    let mut e2_blend = RVector::zeros(kpc.nlam);
765    let mut e_in = CVector::zeros(kpc.ncore);
766
767    // Mixing, for all wavelengths
768    for il in 0..kpc.nlam {
769        // Core
770        if kpc.nmat == 1 && kpc.pcore == 0.0 {
771            // Solid core, single material, nothing to blend for the core
772            e1_blend[il] = e1[[0, il]];
773            e2_blend[il] = e2[[0, il]];
774        } else {
775            // Blend the core materials
776            for im in 0..kpc.ncore {
777                e_in[im] = Complex::new(e1[[il, im]], e2[[il, im]]);
778            }
779            let e_out = bruggeman_blend(
780                &kpc.materials
781                    .iter()
782                    .filter(|m| matches!(m.kind, MaterialKind::Core))
783                    .map(|m| m.vfrac)
784                    .collect::<Vec<_>>(),
785                &e_in.to_vec(),
786            )?;
787            e1_blend[il] = e_out.re;
788            e2_blend[il] = e_out.im;
789        }
790
791        // Mantle
792        if kpc.nmant > 0 {
793            // We do have a mantle to add
794            if (kpc.nmant == 1) && (kpc.pmantle == 0.0) {
795                // No Blending needed inside the mantle - just copy e1 and e2
796                // Since it is only one material, we know it is index nm
797                e1_blend[il] = e1[[kpc.nmat, il]];
798                e2_blend[il] = e2[[kpc.nmat, il]];
799            } else {
800                // Blend the mantle materials
801                for im in 0..kpc.nmant {
802                    e_in[im] = Complex::new(e1[[il, im + kpc.ncore]], e2[[il, im + kpc.ncore]]);
803                }
804                let e_out = bruggeman_blend(
805                    &kpc.materials
806                        .iter()
807                        .filter(|m| matches!(m.kind, MaterialKind::Mantle))
808                        .map(|m| m.vfrac)
809                        .collect::<Vec<_>>(),
810                    &e_in.to_vec(),
811                )?;
812                e1mantle[il] = e_out.re;
813                e1mantle[il] = e_out.im;
814            }
815            let (e1mg, e2mg) = maxwell_garnet_blend(
816                Complex::new(e1_blend[il], e2_blend[il]), // Core
817                Complex::new(e1mantle[il], e2mantle[il]), // Mantle
818                kpc.tot_vfrac_mantle,
819            );
820            e1_blend[il] = e1mg;
821            e2_blend[il] = e2mg;
822        }
823    }
824
825    if kpc.blend_only {
826        // Write .lnk file
827        todo!()
828    }
829    // TODO: Now there is some `--print` command logic
830    // TODO: that we will implement later
831
832    if kpc.write_grid || kpc.blend_only {
833        println!("Exiting after writing requested files");
834        exit(0);
835    }
836
837    // Check how we are going to average over hollow sphere components
838    let (f, wf) = if nf > 1 && kpc.fmax > 0.01 {
839        // Get the weights for Gauss-Legendre integration
840        gauss_legendre(0.01f64, kpc.fmax, nf)
841    } else if kpc.fmax == 0.0 {
842        (RVector::zeros(nf), RVector::ones(nf))
843    } else {
844        // Just a compact sphere, weight is 1
845        (RVector::ones(nf) * kpc.fmax, RVector::ones(nf))
846    };
847
848    // Initialize mu
849    let mut mu = RVector::zeros(kpc.nang);
850    let nangby2f = (kpc.nang / 2) as f64;
851    for (j, val) in mu.iter_mut().take(kpc.nang / 2).enumerate() {
852        let jf = j as f64;
853        let theta = (jf + 0.5) / nangby2f * PI / 2.0;
854        *val = theta.cos();
855    }
856
857    // Create shared memory for the variables
858    let kps = KappaState {
859        ns,
860        nf,
861        ifmn,
862        r: &r,
863        nr: &nr,
864        f: &f,
865        wf: &wf,
866        e1_blend: &e1_blend,
867        e2_blend: &e2_blend,
868        mu: &mu,
869    };
870
871    let p = {
872        let bar = ProgressBar::new(kpc.nlam as u64);
873        bar.set_style(
874            ProgressStyle::default_bar()
875                .template("{spinner} [{elapsed_precise}] [{bar:40.cyan/blue}] {pos}/{len} ({eta})")
876                .unwrap()
877                .progress_chars("█▓▒░ "),
878        );
879        let counter = Arc::new(AtomicUsize::new(0));
880        let results: Result<Vec<KappaResult>, _> = if !kpc.split {
881            let counter = Arc::clone(&counter);
882            let bar = bar.clone();
883            let res = (0..kpc.nlam)
884                .into_par_iter()
885                .map(|ilam| {
886                    let r = over_wavelengths(ilam, &kps, kpc);
887                    let prev = counter.fetch_add(1, AtomicOrdering::SeqCst);
888                    bar.set_position((prev + 1) as u64);
889                    r
890                })
891                .collect();
892            bar.finish_with_message("Processing complete!");
893            res
894        } else {
895            (0..kpc.nlam)
896                .map(|ilam| over_wavelengths(ilam, &kps, kpc))
897                .collect()
898        };
899
900        results.map(|lamrs| {
901            let rho = lamrs[0].mass / lamrs[0].vol;
902            let k_ext = lamrs.iter().map(|lamr| lamr.k_ext).collect::<RVector>();
903            let k_sca = lamrs.iter().map(|lamr| lamr.k_sca).collect::<RVector>();
904            let k_abs = lamrs.iter().map(|lamr| lamr.k_abs).collect::<RVector>();
905            let g = lamrs.iter().map(|lamr| lamr.g).collect::<RVector>();
906            let trust = lamrs.iter().map(|lamr| lamr.trust).collect::<BVector>();
907            let mueller = lamrs
908                .into_iter()
909                .map(|lamr| lamr.mueller)
910                .collect::<Vec<Mueller>>();
911
912            let mut is_ok = true;
913            let mut is_ok_lmin = 0.0;
914
915            for ilam in 0..kpc.nlam {
916                if !trust[ilam] {
917                    is_ok = false;
918                    is_ok_lmin = kpc.lam[ilam];
919                };
920            }
921
922            Particle {
923                rho,
924                k_abs,
925                k_ext,
926                k_sca,
927                g,
928                f: mueller,
929                trust,
930                is_ok,
931                is_ok_lmin,
932                ..Default::default()
933            }
934        })?
935    };
936
937    Ok(p)
938}

940fn bruggeman_blend(abun: &[f64], e_in: &[Complex64]) -> Result<Complex64> {
941    let mut abunvac = 1.0 - abun.iter().sum::<f64>();
942    let mvac = Complex::new(1.0, 0.0);
943    if abunvac < 0.0 {
944        if abunvac.abs() < 1e-5 {
945            abunvac = 0.0;
946        } else {
947            return Err(anyhow!("UnnormalizedAbundance"));
948        }
949    }
950
951    // Blend iteratively
952    let mut mm = mvac;
953    let mut tot: Complex64 = Complex::new(0.0, 0.0);
954    let mut me: Complex64 = Complex::new(0.0, 0.0);
955    for _ in 0..100 {
956        tot = ((mvac.powi(2) - mm.powi(2)) / (mvac.powi(2) + 2.0 * mm.powi(2))) * abunvac;
957        tot += e_in
958            .iter()
959            .zip(abun.iter())
960            .map(|(&e_in_j, &abun_j)| {
961                ((e_in_j.powi(2) - mm.powi(2)) / (e_in_j.powi(2) + 2.0 * mm.powi(2))) * abun_j
962            })
963            .sum::<Complex64>();
964
965        me = mm * ((2.0 * tot + 1.0) / (1.0 - tot)).sqrt();
966        mm = me;
967    }
968    if tot.abs() / mm.abs() > 1e-6 {
969        eprintln!("WARNING: Bruggeman blend not converged");
970    }
971    Ok(me)
972}

974fn maxwell_garnet_blend(m1: Complex64, m2: Complex64, vf_m: f64) -> (f64, f64) {
975    let vf_c = 1.0 - vf_m;
976    let me = m2.powi(2)
977        * ((2.0 * m2.powi(2) + m1.powi(2) - 2.0 * vf_c * (m2.powi(2) - m1.powi(2)))
978            / (2.0 * m2.powi(2) + m1.powi(2) + vf_c * (m2.powi(2) - m1.powi(2))));
979
980    (me.sqrt().re, me.sqrt().im)
981}

983/// Calculate the opacities for a wavelength value.
984fn over_wavelengths(ilam: usize, kps: &KappaState, kpc: &KappaConfig) -> Result<KappaResult> {
985    let qabsdqext_min = 1e-4;
986    let wvno = 2.0 * PI / kpc.lam[ilam];
987    let mut f11 = RVector::zeros(kpc.nang);
988    let mut f12 = RVector::zeros(kpc.nang);
989    let mut f22 = RVector::zeros(kpc.nang);
990    let mut f33 = RVector::zeros(kpc.nang);
991    let mut f34 = RVector::zeros(kpc.nang);
992    let mut f44 = RVector::zeros(kpc.nang);
993    let mut csca = 0.0;
994    let mut cabs = 0.0;
995    let mut cext = 0.0;
996    let mut smat_nbad = 0;
997    let mut mass = 0.0;
998    let mut vol = 0.0;
999    let nangf = kpc.nang as f64;
1000    let mut mie_f11 = RVector::zeros(kpc.nang);
1001    let mut mie_f12 = RVector::zeros(kpc.nang);
1002    let mut mie_f22 = RVector::zeros(kpc.nang);
1003    let mut mie_f33 = RVector::zeros(kpc.nang);
1004    let mut mie_f34 = RVector::zeros(kpc.nang);
1005    let mut mie_f44 = RVector::zeros(kpc.nang);
1006    for is in 0..kps.ns {
1007        let r1 = kps.r[is];
1008        let mut spheres = false;
1009        let mut too_large = false;
1010        match kpc.method {
1011            KappaMethod::DHS => {
1012                let m_in = Complex::new(1.0, 0.0);
1013                for ifn in 0..kps.nf {
1014                    let mut rad: f64;
1015                    if kps.f[ifn] == 0.0 {
1016                        spheres = true;
1017                    } else if r1 * wvno > kpc.xlim {
1018                        too_large = true;
1019                    } else {
1020                        rad = r1 / (1.0 - kps.f[ifn]).powf(1.0 / 3.0);
1021                        let rcore = rad * kps.f[ifn].powf(1.0 / 3.0);
1022                        let mconj = Complex::new(kps.e1_blend[ilam], -kps.e2_blend[ilam]);
1023                        let wvno_1 = wvno.min(kpc.xlim_dhs / rad);
1024                        let dhsc = DHSConfig {
1025                            r_core: rcore,
1026                            r_shell: rad,
1027                            wave_number: wvno_1,
1028                            r_indsh: mconj,
1029                            r_indco: m_in,
1030                            mu: kps.mu,
1031                            numang: kpc.nang / 2,
1032                            max_angle: kpc.nang,
1033                        };
1034                        let (csmie, mut cemie) = match toon_ackerman_1981(&dhsc) {
1035                            Ok(dhsr) => {
1036                                if spheres {
1037                                    rad = r1;
1038                                } else if too_large {
1039                                    rad = r1 / (1.0 - kps.f[kps.ifmn]).powf(1.0 / 3.0);
1040                                };
1041                                let cemie = dhsr.q_ext * PI * rad.powi(2);
1042                                let csmie = dhsr.q_sca * PI * rad.powi(2);
1043                                let factor = 2.0 * PI / csmie / wvno.powi(2);
1044                                for j in 0..(kpc.nang / 2) {
1045                                    mie_f11[j] = (dhsr.m1[[j, 0]] + dhsr.m0[[j, 0]]) * factor;
1046                                    mie_f12[j] = (dhsr.m1[[j, 0]] - dhsr.m0[[j, 0]]) * factor;
1047                                    mie_f22[j] = (dhsr.m1[[j, 0]] + dhsr.m0[[j, 0]]) * factor;
1048                                    mie_f33[j] = (dhsr.s10[[j, 0]]) * factor;
1049                                    mie_f34[j] = (-dhsr.d10[[j, 0]]) * factor;
1050                                    mie_f44[j] = (dhsr.s10[[j, 0]]) * factor;
1051                                    // Here we use the assumption that the grid is regular.  An adapted
1052                                    // grid is not possible if it is not symmetric around pi/2.
1053                                    mie_f11[kpc.nang - j - 1] =
1054                                        (dhsr.m1[[j, 1]] + dhsr.m0[[j, 1]]) * factor;
1055                                    mie_f12[kpc.nang - j - 1] =
1056                                        (dhsr.m1[[j, 1]] - dhsr.m0[[j, 1]]) * factor;
1057                                    mie_f22[kpc.nang - j - 1] =
1058                                        (dhsr.m1[[j, 1]] + dhsr.m0[[j, 1]]) * factor;
1059                                    mie_f33[kpc.nang - j - 1] = (dhsr.s10[[j, 1]]) * factor;
1060                                    mie_f34[kpc.nang - j - 1] = (-dhsr.d10[[j, 1]]) * factor;
1061                                    mie_f44[kpc.nang - j - 1] = (dhsr.s10[[j, 1]]) * factor;
1062                                }
1063                                // exit(0);
1064                                (csmie, cemie)
1065                            }
1066                            Err(e) => {
1067                                println!("DHS error: {:?}", e);
1068                                rad = r1;
1069                                let rmie = rad;
1070                                let lmie = kpc.lam[ilam];
1071                                let e1_mie = kps.e1_blend[ilam];
1072                                let e2_mie = kps.e2_blend[ilam];
1073
1074                                let thmin = 180.0 * (1.0 - 0.5) / nangf;
1075                                let thmax = 180.0 * (nangf - 0.5) / nangf;
1076                                let radius = if rmie / lmie < 5000.0 {
1077                                    rmie
1078                                } else {
1079                                    rmie / 5000.0
1080                                };
1081                                let miec = MieConfig {
1082                                    nangle: kpc.nang,
1083                                    delta: 1e-8,
1084                                    thmin,
1085                                    thmax,
1086                                    step: (thmax - thmin) / (nangf - 1.0),
1087                                    lam: lmie,
1088                                    cmm: Complex::new(e1_mie, e2_mie),
1089                                    rad: radius,
1090                                };
1091                                let mier = de_rooij_1984(&miec)?;
1092                                let cemie = mier.c_ext;
1093                                let csmie = mier.c_sca;
1094                                (csmie, cemie)
1095                            }
1096                        };
1097
1098                        let mut tot = 0.0;
1099                        let mut tot2 = 0.0;
1100                        for j in 0..kpc.nang {
1101                            let jf = j as f64;
1102                            tot += mie_f11[j] * (PI * (jf + 0.5) / nangf).sin();
1103                            tot2 += (PI * (jf + 0.5) / nangf).sin();
1104                        }
1105                        mie_f11[0] += (tot2 - tot) / (PI * 0.5 / nangf).sin();
1106                        if mie_f11[0] < 0.0 {
1107                            mie_f11[0] = 0.0;
1108                        }
1109                        for j in 0..kpc.nang {
1110                            f11[j] += kps.wf[ifn] * kps.nr[is] * mie_f11[j] * csmie;
1111                            f12[j] += kps.wf[ifn] * kps.nr[is] * mie_f12[j] * csmie;
1112                            f22[j] += kps.wf[ifn] * kps.nr[is] * mie_f22[j] * csmie;
1113                            f33[j] += kps.wf[ifn] * kps.nr[is] * mie_f33[j] * csmie;
1114                            f34[j] += kps.wf[ifn] * kps.nr[is] * mie_f34[j] * csmie;
1115                            f44[j] += kps.wf[ifn] * kps.nr[is] * mie_f44[j] * csmie;
1116                        }
1117                        let mut camie = cemie - csmie;
1118                        if camie < cemie * qabsdqext_min {
1119                            camie *= 1e-4;
1120                            cemie = camie + csmie;
1121                        }
1122                        cext += kps.wf[ifn] * kps.nr[is] * cemie;
1123                        csca += kps.wf[ifn] * kps.nr[is] * csmie;
1124                        cabs += kps.wf[ifn] * kps.nr[is] * camie;
1125                        mass += kps.wf[ifn] * kps.nr[is] * kpc.rho_av * 4.0 * PI * r1.powi(3) / 3.0;
1126                        vol += kps.wf[ifn] * kps.nr[is] * 4.0 * PI * r1.powi(3) / 3.0;
1127                    }
1128                }
1129            }
1130            KappaMethod::MMF => {
1131                let m_mono = 4.0 * PI / 3.0 * kpc.mmf_a0.powi(3) * kpc.rho_av;
1132                let v_agg = 4.0 * PI / 3.0 * r1.powi(3);
1133                let m_agg = v_agg * kpc.rho_av;
1134                let nmono = m_agg / m_mono;
1135                let dfrac = if kpc.mmf_struct > 1.0 {
1136                    kpc.mmf_struct
1137                } else {
1138                    3.0 * nmono.ln() / (nmono - kpc.mmf_struct).ln()
1139                };
1140                let kfrac = if kpc.mmf_kf > 0.0 {
1141                    kpc.mmf_kf
1142                } else {
1143                    (5.0 / 3.0).powf(dfrac)
1144                };
1145                let fracc = FractalConfig {
1146                    solver: FractalSolver::ModifiedMeanField,
1147                    cutoff: FractalCutoff::Gaussian,
1148                    geometry: FractalGeometry::Tazaki,
1149                    nang: kpc.nang / 2 + 1,
1150                    pn: nmono,
1151                    r0: kpc.mmf_a0,
1152                    df: dfrac,
1153                    k0: kfrac,
1154                    lmd: kpc.lam[ilam],
1155                    refrel: Complex::new(kps.e1_blend[ilam], kps.e2_blend[ilam]),
1156                    ..Default::default()
1157                };
1158                let fracr = mean_scattering(&fracc)?;
1159                if fracr.dphi > 1.0 {
1160                    smat_nbad += 1;
1161                }
1162                let factor = 4.0 * PI / wvno.powi(2);
1163                for j in 0..kpc.nang {
1164                    f11[j] +=
1165                        kps.nr[is] * 0.5 * (fracr.smat[[0, j]] + fracr.smat[[0, j + 1]]) * factor;
1166                    f12[j] +=
1167                        kps.nr[is] * 0.5 * (fracr.smat[[1, j]] + fracr.smat[[1, j + 1]]) * factor;
1168                    f22[j] +=
1169                        kps.nr[is] * 0.5 * (fracr.smat[[0, j]] + fracr.smat[[0, j + 1]]) * factor;
1170                    f33[j] +=
1171                        kps.nr[is] * 0.5 * (fracr.smat[[2, j]] + fracr.smat[[2, j + 1]]) * factor;
1172                    f34[j] +=
1173                        kps.nr[is] * 0.5 * (fracr.smat[[3, j]] + fracr.smat[[3, j + 1]]) * factor;
1174                    f44[j] +=
1175                        kps.nr[is] * 0.5 * (fracr.smat[[2, j]] + fracr.smat[[2, j + 1]]) * factor;
1176                }
1177                cext += kps.nr[is] * fracr.c_ext;
1178                csca += kps.nr[is] * fracr.c_sca;
1179                cabs += kps.nr[is] * fracr.c_abs;
1180                mass += kps.nr[is] * m_agg;
1181                vol += kps.nr[is] * v_agg;
1182            }
1183            KappaMethod::CDE => {
1184                let v = 4.0 * PI * r1.powi(3) / 3.0;
1185                vol += kps.nr[is] * v;
1186                mass += kps.nr[is] * kpc.rho_av * v;
1187                let m = Complex::new(kps.e1_blend[ilam], kps.e2_blend[ilam]);
1188                let dcabs = 2.0 * wvno * v * (m.powi(2) / (m.powi(2) - 1.0) * m.powi(2).ln()).im;
1189                let dcsca = if (kps.e2_blend[ilam] / kps.e1_blend[ilam]).abs() < 1e-6 {
1190                    // Non-absorbing case
1191                    let mre = m.re;
1192                    wvno.powi(4) * v.powi(2) / (3.0 * PI) * (mre.powi(2) - 1.0 - mre.powi(2).ln())
1193                } else {
1194                    // Absorbing case
1195                    wvno.powi(4) * v.powi(2) * (m.powi(2) - 1.0).abs().powi(2)
1196                        / (3.0 * PI * m.powi(2).im)
1197                        * (m.powi(2) / (m.powi(2) - 1.0) * m.powi(2).ln()).im
1198                };
1199                cabs += kps.nr[is] * dcabs;
1200                csca += kps.nr[is] * dcsca;
1201                cext += kps.nr[is] * (dcabs + dcsca);
1202                if is == kps.ns - 1 {
1203                    // Final size, cscat is complete at this point
1204                    // Compute the scattering matrix deep in the Rayleigh limit
1205                    let lmie = kpc.lam[ilam];
1206                    let rmie = lmie / 1e3;
1207                    let e1_mie = kps.e1_blend[kpc.nlam - 1];
1208                    let e2_mie = kps.e2_blend[kpc.nlam - 1];
1209                    let thmin = 180.0 * (1.0 - 0.5) / nangf;
1210                    let thmax = 180.0 * (nangf - 0.5) / nangf;
1211                    let miec = MieConfig {
1212                        nangle: kpc.nang,
1213                        delta: 1e-8,
1214                        thmin,
1215                        thmax,
1216                        step: (thmax - thmin) / (nangf - 1.0),
1217                        lam: lmie,
1218                        cmm: Complex::new(e1_mie, e2_mie),
1219                        rad: rmie,
1220                    };
1221                    let mier = de_rooij_1984(&miec)?;
1222                    let mie_f22 = mier.f_11.clone();
1223                    let mie_f44 = mier.f_33.clone();
1224                    for j in 0..kpc.nang {
1225                        f11[j] = mier.f_11[j] * csca;
1226                        f12[j] = mier.f_12[j] * csca;
1227                        f22[j] = mie_f22[j] * csca;
1228                        f33[j] = mier.f_33[j] * csca;
1229                        f34[j] = mier.f_34[j] * csca;
1230                        f44[j] = mie_f44[j] * csca;
1231                    }
1232                }
1233            }
1234        }
1235    }
1236
1237    // if ilam == 0 {
1238    //     let rho = mass / vol;
1239    // } // TODO: This can be moved out of parallel loop
1240    let mut k_ext = 1e4 * cext / mass;
1241    let mut k_sca = 1e4 * csca / mass;
1242    let k_abs = 1e4 * cabs / mass;
1243    f11 /= csca;
1244    f12 /= csca;
1245    f22 /= csca;
1246    f33 /= csca;
1247    f34 /= csca;
1248    f44 /= csca;
1249
1250    if kpc.chop_angle > 0.0 {
1251        // Flatten the peak
1252        let ichop = (kpc.chop_angle / (180.0 / nangf)) as usize;
1253        if ichop > 1 {
1254            for i in 0..ichop {
1255                f11[i] = f11[ichop + 1];
1256                f12[i] = f12[ichop + 1];
1257                f22[i] = f22[ichop + 1];
1258                f33[i] = f33[ichop + 1];
1259                f34[i] = f34[ichop + 1];
1260                f44[i] = f44[ichop + 1];
1261            }
1262        }
1263        let mut tot = 0.0;
1264        let mut tot2 = 0.0;
1265        for j in 0..kpc.nang {
1266            let jf = j as f64;
1267            tot += f11[j] * (PI * (jf + 0.5) / nangf).sin() * (PI / nangf) * 2.0 * PI;
1268            tot2 += (PI * (jf + 0.5) / nangf).sin() * (PI / nangf) * 2.0 * PI;
1269        }
1270        for j in 0..kpc.nang {
1271            f11[j] = f11[j] * tot2 / tot;
1272            f12[j] = f12[j] * tot2 / tot;
1273            f22[j] = f22[j] * tot2 / tot;
1274            f33[j] = f33[j] * tot2 / tot;
1275            f34[j] = f34[j] * tot2 / tot;
1276            f44[j] = f44[j] * tot2 / tot;
1277        }
1278        k_sca *= tot / tot2;
1279        k_ext = k_sca + k_abs;
1280    }
1281    let mut tot = 0.0;
1282    let mut g = 0.0;
1283    let mut test_scat = true;
1284
1285    if smat_nbad > 0 {
1286        test_scat = false;
1287    }
1288    for i in 0..kpc.nang {
1289        let ir = i as f64;
1290        g += f11[i] * (PI * (ir + 0.5) / nangf).cos() * (PI * (ir + 0.5) / nangf).sin();
1291        tot += f11[i] * (PI * (ir + 0.5) / nangf).sin();
1292    }
1293    g /= tot;
1294    if smat_nbad > 0 && !kpc.mmf_ss {
1295        g = 0.0;
1296        for i in 0..kpc.nang {
1297            f11[i] = 0.0;
1298            f12[i] = 0.0;
1299            f22[i] = 0.0;
1300            f33[i] = 0.0;
1301            f34[i] = 0.0;
1302            f44[i] = 0.0;
1303        }
1304    }
1305
1306    let mueller = Mueller {
1307        f11,
1308        f12,
1309        f22,
1310        f33,
1311        f34,
1312        f44,
1313    };
1314
1315    let lamr = KappaResult {
1316        mueller,
1317        mass,
1318        vol,
1319        k_ext,
1320        k_sca,
1321        k_abs,
1322        g,
1323        trust: test_scat,
1324    };
1325
1326    Ok(lamr)
