47pub fn launch() -> Result<KappaConfig, KappaError> {
48    let mut kpc = KappaConfig::default();
49
50    let mut args = env::args().skip(1).peekable();
51    if args.peek().is_none() {
52        // If no arguments are given, use the DIANA composition
53        kpc = KappaConfig::diana();
54        return Ok(kpc);
55    }
56
57    while let Some(arg) = args.next() {
58        match arg.as_str() {
59            "-c" => {
60                if let Some(material_arg) = args.next() {
61                    match process_material(
62                        &kpc,
63                        &material_arg.to_string(),
64                        &mut args,
65                        MaterialKind::Core,
66                    ) {
67                        Ok(material) => {
68                            kpc.nmat += 1;
69                            kpc.ncore += 1;
70                            kpc.materials.push(material);
71                        }
72                        Err(KappaError::ZeroMassFraction) => {
73                            args.next();
74                        }
75                        Err(e) => return Err(e),
76                    }
77                } else {
78                    return Err(anyhow!(
79                        "Missing material key/file/refractive index path after -c/-m"
80                    )
81                    .into());
82                }
83            }
84            "-m" => {
85                // Same as "-c"
86                if let Some(material_arg) = args.next() {
87                    match process_material(
88                        &kpc,
89                        &material_arg.to_string(),
90                        &mut args,
91                        MaterialKind::Mantle,
92                    ) {
93                        Ok(material) => {
94                            kpc.nmat += 1;
95                            kpc.nmant += 1;
96                            kpc.materials.push(material);
97                        }
98                        Err(KappaError::ZeroMassFraction) => {
99                            args.next();
100                        }
101                        Err(e) => return Err(e),
102                    }
103                } else {
104                    return Err(anyhow!(
105                        "Missing material key/file/refractive index path after -c/-m"
106                    )
107                    .into());
108                }
109            }
110            "--diana" => {
111                kpc = KappaConfig::diana();
112            }
113            "--dsharp" => {
114                kpc = KappaConfig::dsharp();
115            }
116            "--dsharp-no-ice" => {
117                kpc = KappaConfig::dsharp_no_ice();
118            }
119            // Size distribution options
120            "-a" => match process_grain_size(&mut args)? {
121                SizeArg::PowerLaw {
122                    amin,
123                    amax,
124                    apow,
125                    na,
126                } => {
127                    kpc.amin = amin;
128                    if let Some(amax) = amax {
129                        kpc.amax = amax;
130                    }
131                    if let Some(apow) = apow {
132                        kpc.apow = apow;
133                    }
134                    if let Some(na) = na {
135                        kpc.na = na;
136                    }
137                    kpc.sizedis = SizeDistribution::Apow;
138                }
139                SizeArg::LogNormal {
140                    amin,
141                    amax,
142                    amean,
143                    asigma,
144                    na,
145                } => {
146                    kpc.amin = amin;
147                    kpc.amax = amax;
148                    kpc.amean = amean;
149                    kpc.asigma = asigma;
150                    if let Some(na) = na {
151                        kpc.na = na;
152                    }
153                    kpc.sizedis = SizeDistribution::Normal;
154                }
155                SizeArg::File(file) => {
156                    // Read file
157                    (kpc.na, kpc.ameans_file) = read_sizedis_file(&file)?;
158                    kpc.sizedis = SizeDistribution::File;
159                }
160            },
161            "--amin" => {
162                kpc.amin = args
163                    .next()
164                    .ok_or_else(|| anyhow!("Missing argument: --amin"))?
165                    .parse::<f64>()
166                    .map_err(|_| anyhow!("Invalid value type for argument: --amin"))?;
167            }
168            "--amax" => {
169                kpc.amax = args
170                    .next()
171                    .ok_or_else(|| anyhow!("Missing argument: --amax"))?
172                    .parse::<f64>()
173                    .map_err(|_| anyhow!("Invalid value type for argument: --amax"))?;
174            }
175            "--amean" => {
176                kpc.amean = args
177                    .next()
178                    .ok_or_else(|| anyhow!("Missing argument: --amean"))?
179                    .parse::<f64>()
180                    .map_err(|_| anyhow!("Invalid value type for argument: --amean"))?;
181                kpc.sizedis = SizeDistribution::Normal;
182            }
183            "--asig" | "--asigma" => {
184                kpc.asigma = args
185                    .next()
186                    .ok_or_else(|| anyhow!("Missing argument: --asig"))?
187                    .parse::<f64>()
188                    .map_err(|_| anyhow!("Invalid value type for argument: --asig"))?;
189                kpc.sizedis = SizeDistribution::Normal;
190            }
191            "--apow" => {
192                kpc.apow = args
193                    .next()
194                    .ok_or_else(|| anyhow!("Missing argument: --apow"))?
195                    .parse::<f64>()
196                    .map_err(|_| anyhow!("Invalid value type for argument: --apow"))?;
197            }
198            "--na" => {
199                kpc.na = args
200                    .next()
201                    .ok_or_else(|| anyhow!("Missing argument: --na"))?
202                    .parse::<usize>()
203                    .map_err(|_| anyhow!("Invalid value type for argument: --na"))?;
204            }
205            // Wavelength options
206            "-l" => match process_wavelength(&mut args)? {
207                WavelengthArg::Value { lmin, lmax, nlam } => {
208                    kpc.lmin = lmin;
209                    if let Some(lmax) = lmax {
210                        kpc.lmax = lmax;
211                    }
212                    if let Some(nlam) = nlam {
213                        kpc.nlam = nlam;
214                    }
215                    kpc.wavelength_kind = WavelengthKind::Other;
216                }
217                WavelengthArg::File(file) => {
218                    // Read file
219                    let (nlam, lam) = read_wavelength_grid(&file)?;
220                    kpc.nlam = nlam;
221                    let min_lam = lam.iter().reduce(|a, b| if a < b { a } else { b });
222                    let max_lam = lam.iter().reduce(|a, b| if a > b { a } else { b });
223                    match (min_lam, max_lam) {
224                        (Some(min), Some(max)) => {
225                            kpc.lmin = *min;
226                            kpc.lmax = *max;
227                        }
228                        _ => {
229                            return Err(anyhow!("Could not read wavelength grid from file").into());
230                        }
231                    }
232                    kpc.lam = lam;
233                    kpc.wavelength_kind = WavelengthKind::File;
234                }
235            },
236            "--lmin" => {
237                kpc.lmin = args
238                    .next()
239                    .ok_or_else(|| anyhow!("Missing argument: --lmin"))?
240                    .parse::<f64>()
241                    .map_err(|_| anyhow!("Invalid value type for argument: --lmin"))?;
242            }
243            "--lmax" => {
244                kpc.lmax = args
245                    .next()
246                    .ok_or_else(|| anyhow!("Missing argument: --lmax"))?
247                    .parse::<f64>()
248                    .map_err(|_| anyhow!("Invalid value type for argument: --lmax"))?;
249            }
250            "--nlam" => {
251                kpc.nlam = args
252                    .next()
253                    .ok_or_else(|| anyhow!("Missing argument: --nlam"))?
254                    .parse::<usize>()
255                    .map_err(|_| anyhow!("Invalid value type for argument: --nlam"))?;
256            }
257            // Grain geometry and computational method options
258            "-p" | "--porosity" => {
259                kpc.pcore = args
260                    .next()
261                    .ok_or_else(|| anyhow!("Missing argument: --porosity"))?
262                    .parse::<f64>()
263                    .map_err(|_| anyhow!("Invalid value type for argument: --porosity"))?;
264                kpc.pmantle = match args.next() {
265                    Some(arg) => arg.parse::<f64>().map_err(|_| {
266                        anyhow!("Invalid value type for argument for mantle porosity")
267                    })?,
268                    None => kpc.pcore, // Default is to use the same porosity
269                };
270            }
271            "--dhs" | "--fmax" | "--mie" => {
272                kpc.method = KappaMethod::DHS;
273                if arg == "--mie" {
274                    kpc.fmax = 0.0;
275                } else if arg == "--fmax" {
276                    // Peek at the next argument to check if it's a valid number
277                    if let Some(next_arg) = args.peek() {
278                        if let Ok(fmax_value) = next_arg.parse::<f64>() {
279                            args.next(); // Consume the number
280                            kpc.fmax = fmax_value;
281                        } else {
282                            kpc.fmax = 0.8; // Default value when no valid number follows
283                        }
284                    } else {
285                        kpc.fmax = 0.8; // Default value when --fmax is the last argument
286                    }
287                } else {
288                    kpc.fmax = 0.8; // Default value if neither --mie nor --fmax is provided
289                }
290            }
291            "--xlim" => {
292                kpc.xlim = args
293                    .next()
294                    .ok_or_else(|| anyhow!("Missing argument: --xlim"))?
295                    .parse::<f64>()
296                    .map_err(|_| anyhow!("Invalid value type for argument: --xlim"))?;
297            }
298            "--xlim-dhs" => {
299                kpc.xlim_dhs = args
300                    .next()
301                    .ok_or_else(|| anyhow!("Missing argument: --xlim-dhs"))?
302                    .parse::<f64>()
303                    .map_err(|_| anyhow!("Invalid value type for argument: --xlim-dhs"))?;
304            }
305            "--mmf" | "--mmf-ss" => {
306                kpc.method = KappaMethod::MMF;
307                if arg == "--mmf-ss" {
308                    kpc.mmf_ss = true;
309                }
310                let next_arg = args.next();
311                if let Some(next_arg) = next_arg {
312                    if let Ok(mmf_a0) = next_arg.parse::<f64>() {
313                        kpc.mmf_a0 = mmf_a0;
314                        if let Some(next_arg) = args.next() {
315                            if let Ok(mmf_struct) = next_arg.parse::<f64>() {
316                                kpc.mmf_struct = mmf_struct;
317                                if let Some(next_arg) = args.next() {
318                                    kpc.mmf_kf = next_arg
319                                        .parse::<f64>()
320                                        .map_err(|_| anyhow!("Unable to parse `mmf_kf`"))?;
321                                } else {
322                                    kpc.mmf_kf = 0.0;
323                                }
324                            } else {
325                                return Err(anyhow!("Unable to parse `mmf_struct`").into());
326                            }
327                        } else {
328                            kpc.mmf_struct = 0.2; // Default is a filling factor of 20%
329                        }
330                    } else {
331                        return Err(anyhow!("Unable to parse `mmf_a0`").into());
332                    }
333                } else {
334                    kpc.mmf_a0 = 0.1;
335                }
336            }
337            "--cde" => {
338                kpc.method = KappaMethod::CDE;
339            }
340            // Other options
341            "-o" => {
342                if let Some(next_arg) = args.next() {
343                    kpc.outdir = next_arg.to_string();
344                } else {
345                    kpc.outdir = "output".to_string();
346                }
347            }
348            "-s" | "--scat" | "--scatter" => {
349                kpc.write_scatter = true;
350                if let Some(nang) = args.next() {
351                    kpc.nang = nang.parse::<usize>().map_err(|_| {
352                        anyhow!("Unable to parse `nang` argument for -s/--scat/--scatter")
353                    })?;
354                }
355            }
356            "--sp" | "--sparse" => {
357                kpc.write_scatter = true;
358                if let Some(l1) = args.next() {
359                    if let Ok(l1) = l1.parse::<f64>() {
360                        if let Some(l2) = args.next() {
361                            if let Ok(l2) = l2.parse::<f64>() {
362                                kpc.nsparse += 1;
363                                if kpc.nsparse > 10 {
364                                    return Err(anyhow!("Too many sparse files").into());
365                                }
366                                if l1 < l2 {
367                                    kpc.scatlammin[kpc.nsparse - 1] = l1;
368                                    kpc.scatlammax[kpc.nsparse - 1] = l2;
369                                } else {
370                                    kpc.scatlammin[kpc.nsparse - 1] = l2;
371                                    kpc.scatlammax[kpc.nsparse - 1] = l1;
372                                }
373                            } else {
374                                return Err(anyhow!(
375                                    "Parsing error in second argument of --sparse"
376                                )
377                                .into());
378                            }
379                        } else {
380                            // One specific wavelength
381                            kpc.nsparse += 1;
382                            kpc.scatlammin[kpc.nsparse - 1] = l1;
383                        }
384                    } else {
385                        return Err(anyhow!("Parsing error in first argument of --sparse").into());
386                    }
387                } else {
388                    // No arguments
389                    return Err(anyhow!("Missing arguments to --sparse").into());
390                }
391            }
392            "--chop" => {
393                if let Some(chop_angle) = args.next() {
394                    kpc.chop_angle = chop_angle
395                        .parse::<f64>()
396                        .map_err(|_| anyhow!("Unable to parse `chop_angle` argument for --chop"))?;
397                } else {
398                    kpc.chop_angle = 2.0;
399                }
400            }
401            "--radmc" | "--radmc3d" => {
402                todo!()
403            }
404            "--fits" => {
405                kpc.write_fits = true;
406            }
407            "-d" => {
408                kpc.split = true;
409                if let Some(nsub) = args.next() {
410                    kpc.nsubgrains = nsub
411                        .parse::<usize>()
412                        .map_err(|_| anyhow!("Unable to parse `nsub` argument for -d/--split"))?;
413                } else {
414                    kpc.nsubgrains = 1;
415                }
416            }
417            "-w" => {
418                kpc.write_grid = true;
419            }
420            "-b" | "--blend-only" => {
421                kpc.blend_only = true;
422            }
423            // Miscellaneous options
424            "-L" | "--list" => {
425                list_materials();
426                return Ok(kpc);
427            }
428            "-h" => {
429                print_short_help();
430                exit(0);
431            }
432            "--help" => {
433                print_long_help();
434                exit(0);
435            }
436            "-q" | "--quiet" => {
437                todo!()
438            }
439            "-v" | "--version" => {
440                todo!()
441            }
442            // Unknown option
443            _ if arg.starts_with('-') => {
444                println!("Unknown option: {}", arg);
445                return Ok(kpc);
446            }
447            // Positional arguments
448            _ => match process_material(&kpc, arg.as_str(), &mut args, MaterialKind::Core) {
449                Ok(material) => kpc.materials.push(material),
450                Err(KappaError::ZeroMassFraction) => {
451                    args.next();
452                }
453                Err(e) => return Err(e),
454            },
455        }
456    }
457
458    Ok(kpc)
459}

461fn process_grain_size<I>(args: &mut I) -> Result<SizeArg, KappaError>
462where
463    I: Iterator<Item = String>,
464{
465    let amin_str = args
466        .next()
467        .ok_or_else(|| anyhow!("Missing argument: --amin"))?;
468
469    if amin_str.parse::<f64>().is_err() && Path::new(&amin_str).exists() {
470        return Ok(SizeArg::File(amin_str));
471    }
472
473    let mut amin = amin_str
474        .parse::<f64>()
475        .map_err(|_| anyhow!("Invalid value type for argument: --amin"))?;
476
477    let amax_str = args.next();
478    if let Some(amax_str) = amax_str {
479        if let Ok(mut amax) = amax_str.parse::<f64>() {
480            if amax < 0.0 {
481                if amin + amax <= 0.0 {
482                    return Err(anyhow!("Invalid size range: amin + amax <= 0").into());
483                }
484                amin += amax;
485                amax = amin - 2.0 * amax;
486                let apow = 0.0;
487                let na: Option<usize> = None;
488                return Ok(SizeArg::PowerLaw {
489                    amin,
490                    amax: Some(amax),
491                    apow: Some(apow),
492                    na,
493                });
494            }
495            let next_arg = args.next();
496            if let Some(next_arg) = next_arg {
497                if next_arg.contains(':') {
498                    let parts: Vec<&str> = next_arg.split(':').collect();
499                    if parts.len() != 2 {
500                        return Err(
501                            anyhow!("Could not parse centroid size and logarithmic width").into(),
502                        );
503                    }
504                    let amean = parts[0]
505                        .parse::<f64>()
506                        .map_err(|_| anyhow!("Invalid value type for argument `amean`"))?;
507                    let asigma = parts[1]
508                        .parse::<f64>()
509                        .map_err(|_| anyhow!("Invalid value type for argument `asigma`"))?;
510                    let na = args
511                        .next()
512                        .map(|s| {
513                            s.parse::<usize>()
514                                .map_err(|_| anyhow!("Invalid value type for argument `na`"))
515                        })
516                        .transpose()?;
517                    return Ok(SizeArg::LogNormal {
518                        amin,
519                        amax,
520                        amean,
521                        asigma,
522                        na,
523                    });
524                } else if let Ok(apow) = next_arg.parse::<f64>() {
525                    let na = args
526                        .next()
527                        .map(|s| {
528                            s.parse::<usize>()
529                                .map_err(|_| anyhow!("Invalid value type for argument `na`"))
530                        })
531                        .transpose()?;
532                    return Ok(SizeArg::PowerLaw {
533                        amin,
534                        amax: Some(amax),
535                        apow: Some(apow),
536                        na,
537                    });
538                } else {
539                    return Err(anyhow!("Could not parse powerlaw or lognormal arguments").into());
540                }
541            }
542            Ok(SizeArg::PowerLaw {
543                amin,
544                amax: Some(amax),
545                apow: None,
546                na: None,
547            })
548        } else {
549            Err(anyhow!("Error in parsing amax").into())
550        }
551    } else {
552        let amax = amin;
553        let na = 1;
554        Ok(SizeArg::PowerLaw {
555            amin,
556            amax: Some(amax),
557            apow: None,
558            na: Some(na),
559        })
560    }
561}

563fn process_wavelength<I>(args: &mut I) -> Result<WavelengthArg, KappaError>
564where
565    I: Iterator<Item = String>,
566{
567    let lmin_str = args
568        .next()
569        .ok_or_else(|| anyhow!("Missing argument: --lmin"))?;
570
571    if lmin_str.parse::<f64>().is_err() && Path::new(&lmin_str).exists() {
572        return Ok(WavelengthArg::File(lmin_str));
573    }
574
575    let lmin = lmin_str
576        .parse::<f64>()
577        .map_err(|_| anyhow!("Invalid value type for argument: --lmin"))?;
578
579    let lmax_str = args.next();
580    if let Some(lmax_str) = lmax_str {
581        if let Ok(lmax) = lmax_str.parse::<f64>() {
582            let next_arg = args.next();
583            if let Some(next_arg) = next_arg {
584                if let Ok(nlam) = next_arg.parse::<usize>() {
585                    Ok(WavelengthArg::Value {
586                        lmin,
587                        lmax: Some(lmax),
588                        nlam: Some(nlam),
589                    })
590                } else {
591                    Err(anyhow!("Could not parse nlam").into())
592                }
593            } else {
594                let nlam = 300;
595                Ok(WavelengthArg::Value {
596                    lmin,
597                    lmax: Some(lmax),
598                    nlam: Some(nlam),
599                })
600            }
601        } else {
602            Err(anyhow!("Could not parse lmax").into())
603        }
604    } else {
605        let lmax = lmin;
606        let nlam = 1;
607        Ok(WavelengthArg::Value {
608            lmin,
609            lmax: Some(lmax),
610            nlam: Some(nlam),
611        })
612    }
613}

615fn process_material<I>(
616    kpc: &KappaConfig,
617    material_arg: &str,
618    args: &mut Peekable<I>,
619    material_type: MaterialKind,
620) -> Result<Material, KappaError>
621where
622    I: Iterator<Item = String>,
623{
624    let mut material = Material::default();
625    if MATERIAL_KEYS.contains(&material_arg) {
626        material.key = material_arg.to_string();
627        material.kind = material_type;
628        // let component = get_lnk_data(&material.key);
629        // (material.re, material.im) =
630        //     regrid_lnk_data(component.l0, component.n0, component.k0, &kpc.lam, true);
631        // See if there is a next argument
632        // If there is, it should be a mass fraction
633        if let Some(next_arg) = args.peek() {
634            // If it is a number, it is a mass fraction
635            if let Ok(mfrac) = next_arg.parse::<f64>() {
636                material.mfrac = if mfrac == 0.0 {
637                    return Err(KappaError::ZeroMassFraction);
638                } else {
639                    mfrac
640                };
641                args.next();
642                Ok(material)
643            } else {
644                // Next argument is not a number, so it is a material type
645                material.mfrac = 1.0; // Default is to use 100% of the mass
646                Ok(material)
647            }
648        } else {
649            // This material has a default mass fraction of 100%
650            material.mfrac = 1.0;
651            // Move on to the next material
652            Ok(material)
653        }
654    } else if material_arg.ends_with(".lnk") {
655        // Positional argument is a file path
656        // Placeholder for now
657        let rho_in = Option::<f64>::None;
658        let component = read_lnk_file(material_arg, rho_in)?;
659        let l0_slice: &[f64] = &component.l0.to_vec();
660        let n0_slice: &[f64] = &component.n0.to_vec();
661        let k0_slice: &[f64] = &component.k0.to_vec();
662        (material.re, material.im) = regrid_lnk_data(l0_slice, n0_slice, k0_slice, &kpc.lam, true);
663        material.rho = component.rho;
664        material.cmd = RefractiveIndexKind::File;
665        Ok(material)
666    } else if material_arg.contains(':') {
667        // This is n:k:rho format
668        material.cmd = RefractiveIndexKind::CmdLine;
669        let parts: Vec<&str> = material_arg.split(':').collect();
670        if parts.len() != 3 {
671            Err(anyhow!("Invalid material format").into())
672        } else {
673            material.n = parts[0]
674                .parse::<f64>()
675                .map_err(|_| anyhow!("Invalid value type for argument `n`"))?;
676            material.k = parts[1]
677                .parse::<f64>()
678                .map_err(|_| anyhow!("Invalid value type for argument `k`"))?;
679            material.rho = parts[2]
680                .parse::<f64>()
681                .map_err(|_| anyhow!("Invalid value type for argument `rho`"))?;
682            if material.rho <= 0.0 {
683                eprintln!("Density must be positive");
684                return Err(anyhow!("Density must be positive").into());
685            }
686            material.kind = material_type;
687            material.re = material.n * RVector::ones(kpc.nlam);
688            material.im = material.k * RVector::ones(kpc.nlam);
689            Ok(material)
690        }
691    } else {
692        Err(anyhow!("Invalid material key").into())
693    }
694}
