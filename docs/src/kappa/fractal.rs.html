93/// Computes light scattering properties of randomly oriented
94/// fractal dust aggregates by means of the modified mean field theory developed
95/// in [Tazaki & Tanaka (2018)](https://iopscience.iop.org/article/10.3847/1538-4357/aac32d/meta). This code is also capable of computing the light
96/// scattering solution based on the Rayleigh-Gans-Debye theory and the Mean field
97/// theory.
98///
99/// # Arguments
100/// - `iqsca`:  Method switch for the light scatteing solver
101///     - `iqsca = 1`: Rayleigh-Gans-Debye theory
102///     - `iqsca = 2`: Mean field theory
103///     - `iqsca = 3`: Modified mean-field theory
104/// - `iqcor`: Switch for the two-point correction function
105///     - `iqcor = 1`: Gaussian cut-off
106///     - `iqcor = 2`: Exponential cut-off
107///     - `iqcor = 3`: Fractal dimension cut-off
108/// - `iqgeo`: Switch for the geometric cross-section
109///     - `iqgeo = 1`: `π * rc^2`, where `rc` is the characteristic radius.
110///     - `iqgeo = 2`: [Okuzumi et al. (2009)](https://iopscience.iop.org/article/10.1088/0004-637X/698/2/1122/meta)
111///     - `iqgeo = 3`: [Tazaki (2021)](https://ui.adsabs.harvard.edu/abs/2021MNRAS.504.2811T/abstract)
112/// - `nang`: Number of angles `(0, π/2)` for the scattering phase function.
113/// - `pn`: Number of monomers in the aggregate.
114/// - `r0`: Monomer radius.
115/// - `df`: Fractal dimension.
116/// - `k`: Prefactor for the fractal dimension.
117/// - `lmd`: Wavelength of the incident light.
118/// - `refrel`: Complex Refractive index of the monomer.
119///
120/// # Notes
121/// - For `iqsca = 1`:
122///   all outputs would be physically reasonable for the phase shift `< ~1`.
123///
124/// - For `iqsca = 2`:
125///   The extinction cross section would be calculated without limitation.
126///   However, the other outputs would be reliable for the phase shift < ~1.
127///
128/// - For `iqsca = 3`:
129///   The extinction cross section would be calculated without limitation.
130///   Scattering and absorption cross sections could be calculated
131///   for the phase shift > 1, however too large phase shift may
132///   cause some problem.
133///   The asymmetry parameter and the sattering matrix elements would be
134///   reliable for the phase shift < ~1.
135///
136/// - For the two-point correlation function, it is suggested to use `iqcor = 1` because
137///   it is numerically stable and is likely to reproduce optical properties
138///   of fractal aggregates.
139///
140/// # References
141/// - `iqsca = 1`; `iqcor = 1`: [Tazaki et al. (2016), ApJ, 823, 70](https://iopscience.iop.org/article/10.3847/0004-637X/823/2/70/meta)
142/// - `iqsca = 2`: [Botet et al. (1997), ApOpt, 36, 8791](https://opg.optica.org/abstract.cfm?uri=ao-36-33-8791)
143/// - `iqsca = 3`: [Tazaki & Tanaka (2018), ApJ, 860, 79](https://iopscience.iop.org/article/10.3847/1538-4357/aac32d/meta)
144/// - `iqcor = 2`: [Berry & Percival (1986), AcOpt, 33, 577](https://www.tandfonline.com/doi/abs/10.1080/713821987)
145/// - `iqcor = 3`: [Botet et al. (1995), JPhA, 28, 297](https://iopscience.iop.org/article/10.1088/0305-4470/28/2/008)
146/// - `iqgeo = 2`: [Okuzumi et al. (2009), ApJ, 707, 1247](https://iopscience.iop.org/article/10.1088/0004-637X/698/2/1122/meta)
147/// - `iqgeo = 3`: [Tazaki (2021), MNRAS, 504, 2811](https://ui.adsabs.harvard.edu/abs/2021MNRAS.504.2811T/abstract)
148pub fn mean_scattering(fracc: &FractalConfig) -> Result<FractalResult> {
149    let k = 2.0 * PI / fracc.lmd; // Wavenumber
150    let r_g = fracc.r0 * (fracc.pn / fracc.k0).powf(1.0 / fracc.df); // Radius of gyration of the aggregate
151    let r_c = (5.0f64 / 3.0f64).powf(r_g); // Characteristic radius of the aggregate
152    let x_g = k * r_g; // Size parameter of the aggregate
153    let x0 = k * fracc.r0; // Size parameter of the monomer
154
155    let eta = 25.0;
156    let q_r_g_critical = 26.0;
157
158    // Truncation order of the monomer's scattering field
159    let xmax = x0 + 4.0 * x0.powf(1.0 / 3.0) + 2.0;
160    let nmax = xmax as usize;
161
162    if fracc.nang <= 1 {
163        return Err(anyhow!("LowScatteringAngleResolution"));
164    }
165
166    if fracc.pn < 1.0 {
167        return Err(anyhow!("NotEnoughMonomers"));
168    }
169
170    if fracc.df > 3.0 {
171        return Err(anyhow!("ExceedsMaxFractalDimension"));
172    }
173
174    if (2 * nmax) >= 500 {
175        eprintln!("WARNING: The truncation order of monomer's scattered light");
176        eprintln!("         field exceeds the maximum value (=500).          ");
177        eprintln!("         This may cause a code crush at computations of   ");
178        eprintln!("         the spherical Bessel function of the first kind. ");
179    }
180
181    // Compute the phase shift (cf: Eq. 9 in Tazaki & Tanaka 2018)
182    let ff = fracc.pn * (fracc.r0 / r_c).powf(3.0);
183    let mgmref = maxwell_garnett_mixing(fracc.refrel, ff);
184    let dphic = 2.0 * k * r_c * (mgmref - 1.0).norm();
185    let dphi0 = 2.0 * x0 * (fracc.refrel - 1.0).norm();
186    let dphi = dphic.max(dphi0);
187
188    if dphi >= 1.0 {
189        eprintln!("WARNING: dphi >= 1");
190        eprintln!("         The phase shift by an aggregate exceeds unity.");
191        eprintln!("         Output of scattering matrix elements are not  ");
192        eprintln!("         physically guaranteed.                        ");
193    }
194
195    let (an, bn) = lorenz_mie(x0, fracc.refrel, nmax)?;
196
197    let mut ad = CMatrix::zeros((nmax, 2));
198    let mut dd = CMatrix::zeros((nmax, 2));
199
200    ad.slice_mut(s![.., 0]).assign(&an);
201    ad.slice_mut(s![.., 1]).assign(&bn);
202
203    // Solve multiple scattering?
204    match fracc.solver {
205        FractalSolver::RayleighGansDebye => {
206            dd.slice_mut(s![.., 0]).assign(&ad.slice(s![.., 0]));
207            dd.slice_mut(s![.., 1]).assign(&ad.slice(s![.., 1]));
208        }
209        FractalSolver::MeanField | FractalSolver::ModifiedMeanField => {
210            let r = mean_field(fracc, &ad, x_g, nmax)?;
211            for n in 0..nmax {
212                dd[[n, 0]] = r[2 * n - 1];
213                dd[[n, 1]] = r[2 * n];
214            }
215        }
216    }
217
218    let dang = PI / 2.0 / (fracc.nang - 1) as f64;
219    let (s1, s2) = renormalize(fracc, &dd, dang, nmax);
220
221    let mut smat = RMatrix::zeros((2 * fracc.nang, 4));
222    let ang = (0..(2 * fracc.nang))
223        .map(|j| j as f64 * dang)
224        .collect::<RVector>();
225
226    for j in 0..(2 * fracc.nang) {
227        let q = 2.0 * k * (0.5 * ang[j]).sin();
228        let s11 = 0.5 * (s2[j].norm().powi(2) + s1[j].norm().powi(2));
229        let s12 = 0.5 * (s2[j].norm().powi(2) - s1[j].norm().powi(2));
230        let s33 = (s1[j] * s2[j].conj()).re;
231        let s34 = (s2[j] * s1[j].conj()).im;
232        let sq = match fracc.cutoff {
233            FractalCutoff::Gaussian => {
234                if fracc.df == 3.0 {
235                    if q * q * r_g * r_g >= eta {
236                        0.0
237                    } else {
238                        (-q * q * r_g * r_g / 3.0).exp()
239                    }
240                } else {
241                    let al = 0.5 * fracc.df;
242                    let bb = 1.5;
243                    let xx = -q * q * r_g * r_g / fracc.df;
244                    if q * r_g < q_r_g_critical {
245                        special::confluent_hypergeometric(al, bb, xx)
246                    } else {
247                        let cc = PI.sqrt() * fracc.df.powf(0.5 * fracc.df)
248                            / (2.0 * gamma(0.5 * (3.0 - fracc.df)));
249                        cc * (q * r_g).powf(-fracc.df)
250                    }
251                }
252            }
253            FractalCutoff::Exponential => {
254                let xi = (2.0 / (fracc.df * (fracc.df + 1.0))).sqrt() * r_g;
255                if j == 0 {
256                    1.0
257                } else {
258                    ((fracc.df - 1.0) * (q * xi).atan()).sin()
259                        / (q * xi
260                            * (fracc.df - 1.0)
261                            * (1.0 + (q * xi).powi(2)).powf((fracc.df - 1.0) / 2.0))
262                }
263            }
264            FractalCutoff::FractalDimension => structure_factor_integration(fracc.df, q, r_g),
265        };
266
267        smat[[j, 0]] = fracc.pn * s11 * (1.0 + (fracc.pn - 1.0) * sq);
268        smat[[j, 1]] = fracc.pn * s12 * (1.0 + (fracc.pn - 1.0) * sq);
269        smat[[j, 2]] = fracc.pn * s33 * (1.0 + (fracc.pn - 1.0) * sq);
270        smat[[j, 3]] = fracc.pn * s34 * (1.0 + (fracc.pn - 1.0) * sq);
271
272        if smat[[j, 0]] < 0.0 {
273            return Err(anyhow!("IllegalMatrixElement"));
274        }
275    }
276
277    let mut c_sca: f64 = (0..(fracc.nang - 1))
278        .map(|j| {
279            dang * (smat[[j, 0]] * (ang[j].sin())
280                + 4.0 * smat[[j + 1, 0]] * (ang[j + 1].sin())
281                + smat[[j + 2, 0]] * (ang[j + 2].sin()))
282                / 3.0
283        })
284        .sum::<f64>()
285        * 2.0
286        * PI
287        / (k * k);
288
289    let pf = smat.slice(s![.., 0]).to_owned() / c_sca / (k * k);
290    let g = (0..(fracc.nang - 1))
291        .map(|j| {
292            dang * (pf[j] * (ang[j].sin()) * (ang[j].cos())
293                + 4.0 * pf[j + 1] * (ang[j + 1].sin()) * (ang[j + 1].cos())
294                + pf[j + 2] * (ang[j + 2].sin()) * (ang[j + 2].cos()))
295                / 3.0
296        })
297        .sum::<f64>()
298        * 2.0
299        * PI;
300    let nrm = (0..(fracc.nang - 1))
301        .map(|j| {
302            dang * (pf[j] * (ang[j].sin())
303                + 4.0 * pf[j + 1] * (ang[j + 1].sin())
304                + pf[j + 2] * (ang[j + 2].sin()))
305                / 3.0
306        })
307        .sum::<f64>()
308        * 2.0
309        * PI;
310
311    // Check normalization of phase function
312    if (nrm - 1.0).abs() > 1e-3 {
313        return Err(anyhow!("UnnormalizedPhaseFunction"));
314    }
315    let c_abs: f64;
316    let c_ext: f64;
317    (c_sca, c_abs, c_ext) = match fracc.solver {
318        FractalSolver::RayleighGansDebye => {
319            let c_abs = (0..nmax)
320                .map(|j| {
321                    let jf = j as f64;
322                    let cn1 = 1.0 / ad[[j, 0]].conj() - 1.0;
323                    let cn2 = 1.0 / ad[[j, 1]].conj() - 1.0;
324
325                    (2.0 * jf + 3.0)
326                        * (cn1 * ad[[j, 0]].norm() * ad[[j, 0]].norm()
327                            + cn2 * ad[[j, 1]].norm() * ad[[j, 1]].norm())
328                        .re
329                })
330                .sum::<f64>()
331                * 2.0
332                * PI
333                * fracc.pn
334                / k
335                / k;
336            let c_ext = c_abs + c_sca;
337            (c_sca, c_abs, c_ext)
338        }
339        FractalSolver::MeanField => {
340            let c_ext = (0..nmax)
341                .map(|j| {
342                    let jf = j as f64;
343                    (2.0 * jf + 3.0) * (dd[[j, 0]] + dd[[j, 1]]).re
344                })
345                .sum::<f64>()
346                * 2.0
347                * PI
348                * fracc.pn
349                / k
350                / k;
351            let c_abs = c_ext - c_sca;
352            (c_sca, c_abs, c_ext)
353        }
354        FractalSolver::ModifiedMeanField => {
355            let c_ext = (0..nmax)
356                .map(|j| {
357                    let jf = j as f64;
358                    (2.0 * jf + 3.0) * (dd[[j, 0]] + dd[[j, 1]]).re
359                })
360                .sum::<f64>()
361                * 2.0
362                * PI
363                * fracc.pn
364                / k
365                / k;
366            let mut c_abs = (0..nmax)
367                .map(|j| {
368                    let cn1 = 1.0 / ad[[j, 0]].conj() - 1.0;
369                    let cn2 = 1.0 / ad[[j, 1]].conj() - 1.0;
370                    (2.0 * j as f64 + 3.0)
371                        * (cn1 * ad[[j, 0]].norm() * ad[[j, 0]].norm()
372                            + cn2 * ad[[j, 1]].norm() * ad[[j, 1]].norm())
373                        .re
374                })
375                .sum::<f64>()
376                * 2.0
377                * PI
378                * fracc.pn
379                / k
380                / k;
381
382            let gc = match fracc.geometry {
383                FractalGeometry::Circular => PI * r_c * r_c,
384                FractalGeometry::Okuzumi => {
385                    get_geometric_cross_section_okuzumi(fracc.pn, fracc.r0, r_c)
386                }
387                FractalGeometry::Tazaki => {
388                    get_geometric_cross_section_tazaki(
389                        &IntegrationMethod::AnAn,
390                        &AFactor::Unity,
391                        &fracc.cutoff,
392                        fracc.pn,
393                        fracc.k0,
394                        fracc.df,
395                    )? * fracc.pn
396                        * PI
397                        * fracc.r0
398                        * fracc.r0
399                }
400            };
401            let tau = c_abs / gc;
402            if tau >= 10.0 {
403                c_abs = gc;
404            } else {
405                c_abs = gc * (1.0 - (-tau).exp());
406            }
407            c_abs = c_abs.max(c_ext - c_sca);
408            c_sca = c_ext - c_abs;
409            (c_sca, c_abs, c_ext)
410        }
411    };
412
413    let fracr = FractalResult {
414        c_ext,
415        c_sca,
416        c_abs,
417        g,
418        dphi,
419        ang,
420        smat,
421        pf,
422    };
423    Ok(fracr)
424}

426/// This function finds effective refractive index of an aggregates
427/// by using Maxwell-Garnett mixing rule.
428///
429/// # Arguments
430/// - `refrel`: Complex refractive index of the monomer.
431/// - `f1`: Volume filling factor of the first material.
432///
433/// # Returns
434/// - Effective refractive index of the aggregates.
435///
436/// # Notes
437/// - `eps_1` : Dielectric constant of the first material.
438/// - `eps_2` : Dielectric constant of vacuum.
439/// - `f1` : Volume filling factor of the first material.
440fn maxwell_garnett_mixing(refrel: Complex64, f1: f64) -> Complex64 {
441    let eps_1 = refrel * refrel;
442    let eps_2 = Complex::new(1.0, 0.0);
443    let mg = eps_2 * (2.0 * f1 * (eps_1 - eps_2) + eps_1 + 2.0 * eps_2)
444        / (eps_1 + 2.0 * eps_2 - f1 * (eps_1 - eps_2));
445
446    mg.sqrt()
447}

449/// This subroutine performs integration of the static structure factor by
450/// Botet et al. (1995) based on  `df`-dependent cut-off function for two-point
451/// correlation function.
452/// The statis structure factor `S(q)` is given by
453///
454///
455/// $$
456/// S(q) = \frac{c \cdot df}{q R_g} \int_0^{x_{\max}} dx \, x^{df-2} \sin(q R_g x) e^{-c x^{df}}
457/// $$
458///
459/// where `x = u/Rg`; where `Rg` is the radius of gyration and `u` is the
460/// distance between two monomers; `q` is the magnitude of scattering
461/// vector, `df` is the fractal dimension, `c=0.5` is normalization factor.
462///
463/// Since we have a steep cut-off function for `x>1`, it would be enough
464/// to take `x_max` slightly larger than unity.
465/// Here we determine xmax so that `eta=c*x^df` becomes 25.
466///
467/// We use Simpson's rule for the numerical integration, and then,
468/// the integration range `[0,xmax]` will be divided by `2*m` points.
469///
470/// # Arguments
471/// - `d`: Fractal dimension.
472/// - `q`: Magnitude of scattering vector.
473/// - `r_g`: Radius of gyration.
474///
475/// # Returns
476/// - Value of the static structure factor.
477///
478/// # Caution
479/// For large `df(>2)` and large `qRg`, the numerical integration sometimes
480/// returns a negative value of `S(q)`. Since `S(q)` is a power spectrum,
481/// it must be positive. I also noticed that the same problem arises with
482/// the code written by Pascal Rannou. Because of this ill behavior, I
483/// prefer to use Gausian-cut-off model, which is more stable.
484///
485/// A not-too-bad tentative prescription is simply set `S(q)=0` when `S(q)<0`,
486/// which means scattered waves from each monomer is incoherent. This is
487/// what I did in if-sentence at the end of this subroutine.
488fn structure_factor_integration(d: f64, q: f64, r_g: f64) -> f64 {
489    let integration_grid_size = 100_000;
490    let normalization_factor = 0.5; // Botet+ 1997
491
492    // Upper limit of integration
493    let eta: f64 = 25.0;
494    let xmax = (eta / normalization_factor).powf(1.0 / d);
495
496    let mut s_q: f64;
497    if q == 0.0 {
498        s_q = 1.0;
499    } else {
500        s_q = 0.0;
501        let h = xmax / (2.0 * integration_grid_size as f64);
502        for i in 0..integration_grid_size - 2 {
503            let ir = i as f64;
504            let x0 = 2.0 * ir * h;
505            let x1 = (2.0 * ir + 1.0) * h;
506            let x2 = (2.0 * ir + 2.0) * h;
507            s_q += h
508                * (structure_factor_fn(x0, normalization_factor, d, q, r_g)
509                    + 4.0 * structure_factor_fn(x1, normalization_factor, d, q, r_g)
510                    + structure_factor_fn(x2, normalization_factor, d, q, r_g))
511                / 3.0;
512        }
513        s_q *= normalization_factor * d / q / r_g;
514    }
515    if s_q < 0.0 {
516        s_q = 0.0;
517    }
518
519    s_q
520}

522/// Integrand of the static structure factor.
523///
524/// # Arguments
525/// - `x`: Integration variable.
526/// - `c`: Normalization factor.
527/// - `d`: Fractal dimension.
528/// - `q`: Magnitude of scattering vector.
529/// - `r_g`: Radius of gyration.
530///
531/// # Returns
532/// - Value of the integrand.
533fn structure_factor_fn(x: f64, c: f64, d: f64, q: f64, r_g: f64) -> f64 {
534    if x == 0.0 {
535        0.0
536    } else {
537        let val = (q * r_g * x).sin() * (-c * c.powf(d)).exp() * x.powf(d - 2.0);
538        if val.abs() <= 1e-30 {
539            0.0
540        } else {
541            val
542        }
543    }
544}

546/// Calculate Lorenz-Mie scattering coefficients (an,bn) for a monomer particle.
547///
548/// Since monomer's size parameter is supposed not to be very large,
549/// we use the simple Bohren & Huffman Mie algorithm.
550///
551/// # Arguments
552/// - `x`: Size parameter.
553/// - `refrel`: Complex refractive index of the monomer.
554/// - `nstop`: Number of terms in the series expansion.
555///
556/// # Returns
557/// - `a`: Scattering coefficient.
558/// - `b`: Absorption coefficient.
559///
560/// # References
561/// The original BHMIE code is taken from Bruce Draine's homepage:
562/// <https://www.astro.princeton.edu/~draine/scattering.html>, with
563/// slight modifications.
564fn lorenz_mie(x: f64, refrel: Complex64, nstop: usize) -> Result<(CVector, CVector)> {
565    let nmxx = 150_000;
566
567    let y = refrel * x;
568    let ymod = y.norm();
569    let xstop = x + 4.0 * x.powf(1.0 / 3.0) + 2.0;
570    let nmx = xstop.max(ymod) as usize + 15;
571    if nmx > nmxx {
572        return Err(anyhow!("MieOverflow"));
573    }
574
575    let mut d = CVector::zeros(nmx);
576
577    for n in 0..nmx - 1 {
578        let en = nmx - 1 - n;
579        let enr = (nmx - 1 - n) as f64;
580        d[nmx - 1 - n] = (enr / y) - (1.0 / (d[en] + enr / y))
581    }
582
583    let mut psi_0 = x.cos();
584    let mut psi_1 = x.sin();
585    let mut chi_0 = -psi_1;
586    let mut chi_1 = psi_0;
587    let mut xi_1 = Complex::new(psi_1, -chi_1);
588
589    let mut a = CVector::zeros(nstop);
590    let mut b = CVector::zeros(nstop);
591
592    for n in 0..nstop {
593        let nf = n as f64;
594        let psi = (2.0 * nf - 1.0) * psi_1 / x - psi_0;
595        let chi = (2.0 * nf - 1.0) * chi_1 / x - chi_0;
596        let xi = Complex::new(psi, -chi);
597        a[n] = (d[n] / refrel + nf / x) * psi - psi_1;
598        a[n] /= (d[n] / refrel + (nf / x)) * xi - xi_1;
599        b[n] = (refrel * d[n] + nf / x) * psi - psi_1;
600        b[n] /= (refrel * d[n] + nf / x) * xi - xi_1;
601        psi_0 = psi_1;
602        psi_1 = psi;
603        chi_0 = chi_1;
604        chi_1 = chi;
605        xi_1 = Complex::new(psi_1, -chi_1);
606    }
607    Ok((a, b))
608}

610/// Calculate scattering amplitude S1, S2 from given scattering coefficients d.
611///
612/// # References
613/// The original BHMIE code is taken from Bruce Draine's homepage:
614/// <https://www.astro.princeton.edu/~draine/scattering.html>, with
615/// slight modifications.
616fn renormalize(fracc: &FractalConfig, d: &CMatrix, dang: f64, nmax: usize) -> (CVector, CVector) {
617    let amu = (0..fracc.nang)
618        .map(|j| j as f64 * dang)
619        .map(|theta| theta.cos())
620        .collect::<RVector>();
621
622    let mut pi = RVector::zeros(fracc.nang);
623    let mut pi0 = RVector::zeros(fracc.nang);
624    let mut pi1 = RVector::zeros(fracc.nang);
625
626    let mut tau = RVector::zeros(fracc.nang);
627
628    let nn = 2 * fracc.nang;
629    let mut s1 = CVector::zeros(nn);
630    let mut s2 = CVector::zeros(nn);
631
632    let mut p = -1.0;
633    // let mut an = Complex::new(0.0, 0.0);
634    // let mut an1 = Complex::new(0.0, 0.0);
635    // let mut bn = Complex::new(0.0, 0.0);
636    // let mut bn1 = Complex::new(0.0, 0.0);
637
638    for n in 0..nmax {
639        let nf = n as f64;
640        let f_n = (2.0 * nf + 3.0) / (nf + 1.0) / (nf + 2.0);
641        // if n > 1 {
642        //     an1 = an;
643        //     bn1 = bn;
644        // }
645        let an = d[[n, 0]];
646        let bn = d[[n, 1]];
647
648        for j in 0..fracc.nang {
649            pi[j] = pi1[j];
650            tau[j] = nf * amu[j] * pi[j] - (nf + 2.0) * pi0[j];
651            s1[j] += f_n * (an * pi[j] + bn * tau[j]);
652            s2[j] += f_n * (an * tau[j] + bn * pi[j]);
653        }
654        p = -p;
655        for j in 0..(fracc.nang - 1) {
656            let jj = 2 * fracc.nang - j - 1;
657            s1[jj] += f_n * p * (an * pi[j] - bn * tau[j]);
658            s2[jj] += f_n * p * (-an * tau[j] + bn * pi[j]);
659        }
660
661        for j in 0..fracc.nang {
662            pi1[j] = ((2.0 * nf + 3.0) * amu[j] * pi[j] - (nf + 2.0) * pi0[j]) / (nf + 1.0);
663            pi0[j] = pi[j];
664        }
665    }
666    (s1, s2)
667}

669/// Solves the mean field theory.
670///
671/// # PART I.
672/// Calculate a(nu,n,p) and b(nu,n,p):
673///
674///                   2p + 1  /+1
675///       a(nu,n,p) = -------- | dx P_nu^1(x) * P_n^1(x) * P_p(x),
676///                      2     /-1
677///     
678///                    2p + 1  /+1                         dP_p(x)
679///       b(nu,n,p) = -------- | dx P_nu^1(x) * P_n^1(x) * -------,
680///                      2     /-1                            dx
681///
682/// where `P_n^m` is the associated Legendre function (`n`: degree, `m`: order),
683///  `P_n` is the Legendre polynominal function
684///  (see Equations (29, 30) in Tazaki & Tanaka 2018).
685///  The integration is performed with the Gauss-Legendre quadrature.
686///
687/// # PART II.
688/// Calculate translation matrix coefficients
689///
690/// Translation coefficients: A and B (Equation # from Tazaki & Tanaka 2018)
691/// T(1,nu,n) : \bar{A}_{1,n}^{1,nu} defined in Equation (14)
692/// T(2,nu,n) : \bar{A}_{1,n}^{1,nu} defined in Equation (15)
693/// anunp     : a(nu,n,p)            defined in Equation (29)
694/// bnunp     : b(nu,n,p)            defined in Equation (30)
695///
696/// Note about the rule of rum with respect to the index p
697/// (RT and Botet, in private communication, 2017).
698///
699/// Just before Equation (4) in Botet et al. (1997), it is written that
700/// p should have the same parity as n+nu. However, this is only true
701/// true for a(nu,n,p), and not for b(nu,n,p)
702/// Thus, in this code, index p runs ALL integers between |nu-n| and nu+n.
703///
704/// Tips for efficient computation.
705/// The parity of the integrand of a and b leads following properties:
706/// - a(nu,n,p) .ne. 0 and b(nu,n,p) = 0 when p has the same parity as n+nu
707/// - b(nu,n,p) .ne. 0 and a(nu,n,p) = 0 when p has the opposite parity to n+nu
708///
709///--------------------------------------------------------------------------------
710fn mean_field(fracc: &FractalConfig, ad: &CMatrix, x_g: f64, nmax: usize) -> Result<CVector> {
711    let jm = 400; // Number of grid points for Gauss-Legendre quadrature
712
713    // Integration limits
714    let x1 = -1.0;
715    let x2 = 1.0;
716
717    // Gauss-Legendre quadrature
718    let (x, w) = legendre::gauss_legendre(x1, x2, jm);
719
720    let order = 1;
721    let pmax = 2 * nmax;
722
723    // Store values of Legendre polynomials (P_n^m) and their derivatives
724    let mut al1n = RMatrix::zeros((jm, nmax));
725    let mut ln = RMatrix::zeros((jm, pmax));
726    let mut dln = RMatrix::zeros((jm, pmax));
727
728    for (j, xj) in x.iter().enumerate() {
729        let (pmn, _) =
730            legendre::lpmns(order, nmax, *xj).context("IntegrationFailure: In Legendre")?;
731        let (lp, dlp) = legendre::lpn(pmax, *xj);
732        al1n.slice_mut(s![j, ..]).assign(&pmn);
733        ln.slice_mut(s![j, ..]).assign(&lp);
734        dln.slice_mut(s![j, ..]).assign(&dlp);
735    }
736
737    // Pre-compute the spherical Bessel functions and the T-matrix
738    let mut s_p = CVector::zeros(pmax);
739
740    for (p, val) in s_p.iter_mut().enumerate() {
741        *val = bessel::int_sph_bessel(fracc, x_g, p).context("IntegrationFailure: In Bessel")?;
742    }
743
744    let mut t = CTensor::zeros((2, nmax, nmax));
745    for nu in 0..nmax {
746        let nuf = nu as f64;
747        for n in 0..nmax {
748            let nf = n as f64;
749            let pmin = nu.abs_diff(n);
750            let pmax = n + nu;
751            let mut sum_a = Complex::new(0.0, 0.0);
752            let mut sum_b = Complex::new(0.0, 0.0);
753            for (p, val) in s_p.iter().enumerate().take(pmax + 1).skip(pmin) {
754                let val = *val;
755                let pf = p as f64;
756                let mut anunp = 0.0;
757                let mut bnunp = 0.0;
758                if pmax % 2 == p % 2 {
759                    anunp = w
760                        .iter()
761                        .zip(al1n.row(nu))
762                        .zip(al1n.row(n))
763                        .zip(ln.row(p))
764                        .map(|(((w, al1n_nu), al1n_n), ln_p)| w * al1n_nu * al1n_n * ln_p)
765                        .sum();
766                } else {
767                    bnunp = w
768                        .iter()
769                        .zip(al1n.row(nu))
770                        .zip(al1n.row(n))
771                        .zip(dln.row(p))
772                        .map(|(((w, al1n_nu), al1n_n), dln_p)| w * al1n_nu * al1n_n * dln_p)
773                        .sum();
774                }
775                anunp *= 0.5 * (2.0 * pf + 1.0);
776                bnunp *= 0.5 * (2.0 * pf + 1.0);
777
778                sum_a += (nf * (nf + 1.0) + nuf * (nuf + 1.0) - pf * (pf + 1.0)) * anunp * val;
779                sum_b += bnunp * val;
780            }
781            t[[0, nu, n]] = sum_a * (2.0 * nuf + 1.0) / (nf * (nf + 1.0) * nuf * (nuf + 1.0));
782            t[[1, nu, n]] = sum_b * 2.0 * (2.0 * nuf + 1.0) / (nf * (nf + 1.0) * nuf * (nuf + 1.0));
783        }
784    }
785
786    let mut s = CMatrix::ones((pmax, 2));
787    let mut y = CVector::zeros(pmax);
788
789    for n in 0..nmax {
790        for nu in 0..nmax {
791            s[[2 * n - 1, 2 * n - 1]] = (fracc.pn - 1.0) * ad[[0, n]] * t[[0, nu, n]];
792            s[[2 * n - 1, 2 * n]] = (fracc.pn - 1.0) * ad[[0, n]] * t[[1, nu, n]];
793            s[[2 * n, 2 * n - 1]] = (fracc.pn - 1.0) * ad[[1, n]] * t[[0, nu, n]];
794            s[[2 * n, 2 * n]] = (fracc.pn - 1.0) * ad[[1, n]] * t[[1, nu, n]];
795        }
796        y[2 * n - 1] = ad[[0, n]];
797        y[2 * n] = ad[[1, n]];
798    }
799
800    // Invert the T-matrix elements S
801    // and find mean-field vector r
802    let r = linalg::complex_matrix_inverse(pmax, 2 * pmax, &y, &s).context("SingularMatrix")?;
803
804    Ok(r)
805}

807fn get_geometric_cross_section_okuzumi(pn: f64, a0: f64, ac: f64) -> f64 {
808    let k0_bcca = 1.04; // Tazaki+2016; BCCA with oblique collisions
809    let df_bcca = 1.90; // Tazaki+2016; BCCA with oblique collisions
810
811    let ac_bcca = (5.0f64 / 3.0f64).sqrt() * a0 * (pn / k0_bcca).powf(1.0 / df_bcca);
812    let gc0 = PI * a0 * a0;
813    let gcc = PI * ac * ac;
814
815    let gc_bcca2 = PI * ac_bcca * ac_bcca;
816
817    let gc_bcca = if pn <= 16.0 {
818        12.5 * pn.powf(0.685) * (-2.53 / pn.powf(0.0920)).exp() * gc0
819    } else {
820        0.352 * pn + 0.566 * pn.powf(0.862) * gc0
821    };
822
823    (1.0 / (1.0 / gc_bcca + 1.0 / gcc - 1.0 / gc_bcca2)).min(pn * PI * a0 * a0)
824}
